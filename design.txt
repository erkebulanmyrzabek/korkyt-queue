function StatusBadge({ status, t }) {
  const map = {
    available: "bg-emerald-50 text-emerald-700 border-emerald-200",
    busy: "bg-amber-50 text-amber-700 border-amber-200",
    offline: "bg-slate-100 text-slate-600 border-slate-200",
    waiting: "bg-blue-50 text-blue-700 border-blue-200",
    active: "bg-fuchsia-50 text-fuchsia-700 border-fuchsia-200",
    completed: "bg-emerald-50 text-emerald-700 border-emerald-200",
  };
  const labels = {
    available: t.available,
    busy: t.busy,
    offline: t.offline,
    waiting: t.waiting,
    active: t.active,
    completed: t.completed,
  };
  return <Badge variant="outline" className={cn("rounded-full px-3 py-1 text-xs font-medium", map[status])}>{labels[status] || status}</Badge>;
}
